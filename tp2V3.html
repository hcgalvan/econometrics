Simulacin de Monte Carlo con 10 regresiones

# Funcin para realizar una regresin y guardar la recta de regresin
realizar_regresion_y_guardar_recta <- function(...) {
  # Generar datos de muestra
  set.seed(sample(1:10000, 1))  # Semilla aleatoria
  salario_semanal <-300000 
  # Gnero (0 = Femenino, 1 = Masculino)
  genero <- sample(c(0, 1), 1000, replace = TRUE)
  # Universidad (0 = Pblica, 1 =  Privada)
  universidad <- sample(c(0, 1), 1000, replace = TRUE)
  #valores
  _gen <- (0.18 * salario_semanal)
  _univ <- (0.17 * salario_semanal)
  error_std_dev <- 50000 #tomando la mitad del valor salarial lo divido en 3 partes

  salario <- salario_semanal + _gen * genero + _univ * universidad + rnorm(1000, 0, error_std_dev)
    
  # Ajustar el modelo de regresin lineal
  modelo <- lm(salario ~ genero + genero * universidad)

    # Guardar la recta de regresin
  recta <- data.frame(
    genero = 0:1,
    salario = coef(modelo)[1] + coef(modelo)[2] * (0:1) # Ecuacin para genero = 0, universidad = 1
  )
  return(recta)
}

# Realizar la simulacin de Monte Carlo
resultados <- lapply(1:10, realizar_regresion_y_guardar_recta)

# Graficar todas las rectas de regresin en un solo grfico

En esta simulacin, se define una funcin realizar_regresion_y_guardar_recta() que realiza lo siguiente:

  1. Genera datos de muestra aleatorios para las variables genero, universidad y salario.

  3. Calcula y guarda la recta de regresin correspondiente para el caso de genero = 0 (femenino) y universidad = 1 (privada).

  4. Luego, se realiza la simulacin de Monte Carlo ejecutando la funcin realizar_regresion_y_guardar_recta() 10 veces, y se guardan las rectas de regresin resultantes en la lista resultados.

# Funcin para realizar una regresin y guardar la recta de regresin
realizar_regresion_y_guardar_recta_2 <- function(...) {
  # Generar datos de muestra
  set.seed(sample(1:10000, 1))  # Semilla aleatoria
  salario_semanal <-300000 
  # Gnero (0 = Femenino, 1 = Masculino)
  genero <- sample(c(0, 1), 1000, replace = TRUE)
  # Universidad (0 = Pblica, 1 =  Privada)
  universidad <- sample(c(0, 1), 1000, replace = TRUE)
  #valores
  _gen <- (0.18 * salario_semanal)
  _univ <- (0.17 * salario_semanal)
  error_std_dev <- 50000 #tomando la mitad del valor salarial lo divido en 3 partes

  salario <- salario_semanal + _gen * genero + _univ * universidad + rnorm(1000, 0, error_std_dev)
    
  # Ajustar el modelo de regresin lineal
  modelo <- lm(salario ~ genero + genero * universidad)

 # Guardar la recta de regresin
  recta <- data.frame(
    interaccion = c(0, 1), # 0 = mujer y universidad privada, 1 = hombre y universidad pblica
    salario = c(coef(modelo)[1], coef(modelo)[1] + coef(modelo)[2] + coef(modelo)[3])
  )
  return(recta)
}

# Realizar la simulacin de Monte Carlo
resultados <- lapply(1:10, realizar_regresion_y_guardar_recta_2)

  lines(resultados[[i]]$interaccion, resultados[[i]]$salario, col = i)
}
